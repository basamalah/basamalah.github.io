[tex]\sf{JO = RS}[/tex]

[tex]\sf{∠P = ∠M = 90°}[/tex]

[tex]\sf{∠J = ∠S}[/tex]

[tex]\boxed{\sf{y = 65°}}[/tex]

[tex]\textsf{Given : Find the x.}[/tex]

[tex]\sf{∠J + ∠P + ∠O = 180 }[/tex]

[tex]\boxed{\sf{x = 25°}}[/tex]

[tex]\textsf{Given : Find the w.}[/tex]

[tex]\sf{∠J + ∠P + ∠O }[/tex]

[tex]\boxed{\sf{w = 25°}}[/tex]
