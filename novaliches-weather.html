[tex] \large \mathcal{ANSWER:} [/tex]

[tex] \boxed{\begin{array}{l} \textsf{Michael} = 9 \textsf{ kph} \\ \textsf{Bongbong} = 10\textsf{ kph} \end{array}} [/tex]

[tex] \large \mathcal{SOLUTION:} [/tex]

[tex] \begin{array}{l} \textsf{“Bongbong traveled 1 km per hour faster than} \\ \:\: \textsf{Michael...”}\\ \\ \textsf{Let }x\textsf{ and }x + 1\textsf{ be the speeds of Michael and} \\ \textsf{Bongbong, in km per hour, respectively.} \\ \\ \textsf{We know that }\textsf{Time} = \dfrac{\textsf{Distance}}{\textsf{Speed}}. \\ \\ \textsf{“...and reached Novaliches 1 hour before him.”} \\ \\ \begin{aligned} \textsf{T}_{\tiny \textsf{Bongbong}} &= \textsf{T}_{\tiny \textsf{Michael}} - 1 \\ \dfrac{90}{x + 1} &= \dfrac{90}{x} - 1 \\ x\cancel{(x + 1)}\left(\dfrac{90}{\cancel{x + 1}}\right) &= \left(\dfrac{90}{x} - 1\right)\!(x)(x +1) \end{aligned} \\ \quad \qquad 90x = 90(x + 1) - x(x + 1) \\ \quad \qquad 90x = 90x + 90 - x^2 - x \\ \begin{aligned} \qquad \qquad x^2 + x - 90 &= 0 \\ \quad \qquad (x + 10)(x - 9) &= 0 \end{aligned} \\ \:\: \qquad \qquad x = -10 \: \Big |\: x = 9\\ \\ \textsf{We only need the positive value of }x, \textsf{ so }x = 9. \\ \\ \textsf{Thus, the rates of Michael and Bongbong are }\\ 9\textsf{ kph and }10\textsf{ kph, respectively.} \end{array} [/tex]
