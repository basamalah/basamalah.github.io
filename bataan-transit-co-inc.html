[tex]\tt\huge\blue{ANSWER}[/tex]

Area = 509

[tex]\tt\huge\green{SOLUTION}[/tex]

50=2w+l

50-2w=l

A=LxW

=(50w-2w)w

=50w-2w²

=-2w²+50

area=-2[w²-25]

=-2[w²-25+12.5-12.5]

=-2[(w-12.5)²-250]

=-2(w-12.5)²+500

Area=500
