[tex]\\ \tt{} - 125 = {r}^{3} [/tex]

[tex]\\ \tt{} \sqrt[3]{ - 125} = r[/tex]

[tex]\\ \tt{} - 5 = r[/tex]

[tex]\\ \tt{}r = - 5[/tex]

============================
